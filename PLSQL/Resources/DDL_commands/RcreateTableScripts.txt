
--Create TABLE Commands

create table pkattep.Rooms
( rtype_id NUMBER(20) NOT NULL,
 floor NUMBER(20) NOT NULL,
 capacity NUMBER(20) NOT NULL,
 --as of now only 2 capacities are assumed - 4 and 10
 roomtype VARCHAR2(10) NOT NULL
 --Type should be one among conf or study
);

create table pkattep.cameras
(rtype_id NUMBER(20) NOT NULL,
 model VARCHAR2(20) NOT NULL,
 memory NUMBER(20),
 lens_config VARCHAR2(20),
 manufacturer VARCHAR2(20) NOT NULL
);

create table pkattep.publications
(rtype_id NUMBER(20) NOT NULL,
 year VARCHAR2(20) NOT NULL,
 title VARCHAR2(50) NOT NULL
);
